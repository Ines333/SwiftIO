DigitalOut

public class DigitalOut

Use the DigitalOut class to set the value of a digital output pin.

Example: A simple hello world.

import SwiftIO

main() {
   //Create a DigitalOut to .D0
   let pin = DigitalOut(.D0)

   //Reverse the output every 1 second
   while true {
       pin.write(1)
       msSleep(1000)
       pin.write(0)
       msSleep(1000)
   }
}
